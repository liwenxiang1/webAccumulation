# 数组解构

```js
var names = ['a', 'b', 'c'];

//对数组进行解构
var [a, b, b] = names;
//结构后两个参数
var [, b, c] = names;
//结构出第一个元素，后面的放数组中
var [a, ...b] = names;
//结构默认值,如果d不复制，则默认为undefined
var [a, b, c, d = 'd'] = names;
```

# 对象结构

```js
var obj = { name: 'a', age: 18 };

//对对象进行解构
var { name, age } = obj;
//结构出name，后面的放数组中
var { name, ...b } = obj;
//结构默认值,如果d不复制，则默认为undefined
var { name, age, d = 'd' } = obj;
// 重命名
var { name: myName, age: myAge } = obj;
```

# 字符串解构

```js
var str = 'abc';
//对字符串进行解构
var [a, b, c] = str;
```

# 函数参数解构

```js
function fn({ name, age }) {
  console.log(name, age);
}
fn({ name: 'a', age: 18 });
```
